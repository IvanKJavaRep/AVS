#include <stdio.h>


int n;
int a[999]; // объявлен массив a из n элементов
int b[998];
int i;
void print_array(int b[])
{
    char str[100] = "\nArray printed";
    // Вывод элементов массива
    for (i = 0; i<n-1; i++)
    {
        printf("%d ", b[i]);
    }
    printf("%s", str);
}
int main()
{
    scanf("%d", &n);
    if(n>=2 && n<1000)
    {
        // Ввод элементов массива
        for (i = 0; i<n; i++)
        {
            scanf("%d", &a[i]);
        }
        for (i = 0; i<n-1; i++)
        {
            b[i]=a[i]+a[i+1];
        }
        print_array(b);
    }else
        printf("для выполнения задачи массив должн состоять минимум из дву элементов, а для исключения использования слишком большого количества ресурсов, ограничим сверху числом 1000");
    return 0;
}
